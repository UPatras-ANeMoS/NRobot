# Libraries to compile
# nbase library from NBase
INCLUDE_DIRECTORIES(NBase)
ADD_SUBDIRECTORY(NBase)
# INCLUDE_DIRECTORIES(NPart)
# ADD_SUBDIRECTORY(NPart)
# INCLUDE_DIRECTORIES(NRobot)
# ADD_SUBDIRECTORY(NRobot)
IF(NR_USE_SDL)
    INCLUDE_DIRECTORIES(NPlot)
    ADD_SUBDIRECTORY(NPlot)
    SET(NR_PLOT_LIB nplot)
ENDIF(NR_USE_SDL)

# Create executables
ADD_EXECUTABLE(NBASE_TEST "TEST_NBase.cpp")
ADD_EXECUTABLE(NPLOT_TEST "TEST_NPlot.cpp")

# Link the libraries
TARGET_LINK_LIBRARIES(NBASE_TEST nbase ${SDL2_LIBRARIES} ${NR_PLOT_LIB})
TARGET_LINK_LIBRARIES(NPLOT_TEST nbase ${SDL2_LIBRARIES} ${NR_PLOT_LIB})

# Set executable output directory and postfix
SET_TARGET_PROPERTIES(NBASE_TEST PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/bin)
SET_TARGET_PROPERTIES(NBASE_TEST PROPERTIES OUTPUT_NAME nbasetest${BUILD_POSTFIX})
SET_TARGET_PROPERTIES(NPLOT_TEST PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/bin)
SET_TARGET_PROPERTIES(NPLOT_TEST PROPERTIES OUTPUT_NAME nplottest${BUILD_POSTFIX})
